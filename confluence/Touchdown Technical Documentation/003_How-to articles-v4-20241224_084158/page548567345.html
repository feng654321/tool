056_How to send 5G traffic on FPC


################  config static route on ei-0    ################.....
fsclish -c 'add route static node ei-0 subnet 44.1.11.0/24 gateway address 25.0.9.1'

################  config static route on FPC    ################.....
ip route add 25.0.9.0/24 via 44.1.11.1 dev traffic1

....................ei-0 ping FPC ..........................
ssh ei-0.local ping 44.1.11.3

....................FPC ping ei-0 ..........................
ping 25.0.9.187

################        Set Fp Pipeline Mode on ei-0    ################.....

/opt/nokia/libexec/trs/rcptrhmappingcli -o del -n pipeline -s ei-1 -v type-1
/opt/nokia/libexec/trs/rcptrhmappingcli -o add -n pipeline -s ei-1 -v type-4
sed -i "s/pipeline = type-1/pipeline = type-4/g" /run/trs/dpdk_conf/local_fp-rte_config.conf
/opt/nokia/libexec/trs/rcptrhmappingcli -o show -n pipeline

#systemctl stop trs_psv_cell

/opt/nokia/bin/FPCiUtils -c 'pipeline upload file /opt/nokia/lib/trs/pipeline/pipeline_ueup.pl'

################        Set l2sim platform on ei-0      ################
/opt/nokia/bin/FPCiUtils -c 'l2sim set platform 5GLatest'
/opt/nokia/bin/FPCiUtils -c 'l2sim show platform'
/opt/nokia/bin/FPCiUtils -c 'l2sim reset runtime'
/opt/nokia/bin/FPCiUtils -c 'l2sim show runtime'
/opt/nokia/bin/FPCiUtils -c 'l2sim stop'
/opt/nokia/bin/FPCiUtils -c 'l2sim reset parameter normal'
/opt/nokia/bin/FPCiUtils -c 'l2sim reset parameter performance'
/opt/nokia/bin/FPCiUtils -c 'l2sim set mode loopback'
/opt/nokia/bin/FPCiUtils -c 'l2sim set parameter local-ip 25.0.9.187;l2sim set parameter peer-ip 44.1.11.3;'
/opt/nokia/bin/FPCiUtils -c 'l2sim show parameter normal'
/opt/nokia/bin/FPCiUtils -c 'l2sim start auto'
/opt/nokia/bin/FPCiUtils -c 'l2sim show mode'

################        up-ip add ip-addr on ei-0       ################
/opt/nokia/bin/FPCiUtils -c 'up-ip add ip-addr 25.0.9.187 backhaul'
/opt/nokia/bin/FPCiUtils -c 'up-ip show'
/opt/nokia/bin/FPCiUtils -c 'classifier show policy table'

################        Udp send traffic on FPC  ################.....
cd /home/jlou/rcp2.0/rcptaf/testcases/TRHRobotTest/robottests/resources/Stub
udpsd
gtool 100 create 44.1.11.3 2152 0xaa000009 25.0.9.187 2152 0x40000013
gtp set message gtpu
ip set length 200
gtool 100 send 5
gtp show stat

################  check point on ei-0 ################
/opt/nokia/bin/FPCiUtils -c 'stat show summary'
/opt/nokia/bin/FPCiUtils -c 'l2sim show runtime'
