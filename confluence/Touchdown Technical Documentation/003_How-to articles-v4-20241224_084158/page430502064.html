031_How to manually run signaling case for mcRNC and vRNC

 

1. mcRNC FT CASE(RU OPE操作)手动执行CASE

1.1 mcRNC上配置如下：

cd /root/crt
./smart_del_dss.sh 
./smart_delete_ip.sh 
source common_rnc_test_env_bashrc_b2
fsclish -f ./ip_and_route_config_b2.sh
./SigtranDss_FT_b2.sh

1.2 FPC上配置如下：

cd21
source  common_pc_test_env_bashrc_ft_b2
./fpc_config_ip_b2.sh
./ss7sim >ss7sim_log&     //这里一定要注意使用正确的ss7sim文件
cd  /home/jlou/SS_ILFT/SS_ILFT/TestCase
vi libpythonpath    //删除前两行后，添加自己下载的CASE目录
    root_path=/home/jlou/SS_ILFT/SS_ILFT/TestCase
    talib_path=/home/jlou/SS_ILFT/SS_ILFT/TALIB/src
cd21;source common_pc_test_env_bashrc_ft_b2   ;source /home/jlou/SS_ILFT/SS_ILFT/TestCase/libpythonpath;cd /home/jlou/SS_ILFT/SS_ILFT/CasePool/Signaling/SCCP/Sigtran_OPE      //注意修改为自己下载的CASE目录
pybot -t "Sigtran_RU_Restart_SP_FSSGWNetMgrServer" Sigtran_RU_Restart.tsv      //执行一个RU OPE的CASE
